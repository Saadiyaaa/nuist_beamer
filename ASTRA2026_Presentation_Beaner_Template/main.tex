% IMPORTANT LAYOUT GUIDE FOR AGENTS AND HUMAN EDITORS
%
% 1. Do not rebuild the image columns manually on text-plus-figure slides.
%    Select the layout by changing the corresponding ...FigureCount value:
%      1 -> one figure fills the complete image board;
%      2 -> two figures are stacked vertically;
%      3 -> three figures occupy three cells of a 2 x 2 grid;
%      4 -> four figures occupy a complete 2 x 2 grid.
%
% 2. Available adaptive boards:
%      Content slide:  \ContentFigureCount
%      Results slide:  \ResultFigureCount
%      Insight slide:  \InsightFigureCount
%
% 3. After choosing the count, fill only fields A--D and their captions.
%    Unused fields are ignored automatically.  Replace a field with an
%    \includegraphics command using a relative path such as
%    \includegraphics{figures/result-a.pdf}, or insert a table/chart/TikZ.
%
% 4. If more than four visuals are needed, duplicate the most suitable slide
%    instead of shrinking every visual until it becomes unreadable.
%
% 5. Preserve the title page, page header, reference placement, and closing
%    page styles.  Split dense prose across slides before reducing font size.
% ============================================================

% count, contents A--D, captions A--D
\newcommand{\AdaptiveFigureBoard}[9]{%
  \ifcase#1
    \ContentFigurePanel{\linewidth}{4.05cm}{[SET FIGURE COUNT TO 1--4]}{}%
  \or
    \ContentFigurePanel{\linewidth}{4.05cm}{#2}{#6}%
  \or
    \ContentFigurePanel{\linewidth}{1.65cm}{#2}{#6}%
    \par\vspace{0.22cm}
    \ContentFigurePanel{\linewidth}{1.65cm}{#3}{#7}%
  \or
    \ContentFigurePanel{0.48\linewidth}{1.45cm}{#2}{#6}\hfill
    \ContentFigurePanel{0.48\linewidth}{1.45cm}{#3}{#7}%
    \par\vspace{0.22cm}
    \ContentFigurePanel{0.48\linewidth}{1.45cm}{#4}{#8}%
  \else
    \ContentFigurePanel{0.48\linewidth}{1.45cm}{#2}{#6}\hfill
    \ContentFigurePanel{0.48\linewidth}{1.45cm}{#3}{#7}%
    \par\vspace{0.22cm}
    \ContentFigurePanel{0.48\linewidth}{1.45cm}{#4}{#8}\hfill
    \ContentFigurePanel{0.48\linewidth}{1.45cm}{#5}{#9}%
  \fi
}
